  try {
    const res = await fetch("http://localhost:8080/users/login", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ username, password }),
    });
    const data: LoginResponse = await res.json();
    return data;
  } catch (error) {
    throw new Error(error);
  }
};
