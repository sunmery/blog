# 列出块设备信息
lsblk

# 查看磁盘的分区和未分配空间
sudo parted /dev/sda print free

# 创建新分区
sudo parted /dev/sda
## 在parted提示符下输入: mkpart <起始> <结束>
mkpart primary 68.7GB 137GB
print 
quit

lsblk
输出:
NAME  MAJ:MIN RM  SIZE RO TYPE MOUNTPOINTS
sda     8:0    0  128G  0 disk 
├─sda1
│       8:1    0    1G  0 part /boot/efi
├─sda2
│       8:2    0    2G  0 part /boot
├─sda3
│       8:3    0 60.9G  0 part 
│ └─ubuntu--vg-ubuntu--lv
│     252:0    0 60.5G  0 lvm  /
└─sda4
sda4 就是新分区的名字

#### 格式化新分区为物理卷
pvcreate /dev/sda4
vgdisplay
vgextend ubuntu-vg /dev/sda4
lvdisplay
lvextend -l +100%FREE  /dev/ubuntu-vg/ubuntu-lv
resize2fs /dev/ubuntu-vg/ubuntu-lv
df -Th /
df -Th /
